## 二、验证流程

### 2.1 登录流程

```
用户输入卡密 → 构建请求参数 → AES加密参数 → 发送HTTP请求 → AES解密响应 → 验证结果
     ↓
成功 → 保存Token → 记住卡密(可选) → 启动心跳线程
失败 → 显示错误信息 → 允许重试
```

### 2.2 心跳保活流程

```
定时执行(默认50秒间隔)
     ↓
发送心跳请求 → 检查响应状态
     ↓
┌─────────────────┬────────────────────┐
│ 正常 (code=1)   │ 异常 (code≠1)      │
│ 重置失败计数    │ 失败计数+1         │
│ 继续心跳        │ 达到阈值则强制退出 │
└─────────────────┴────────────────────┘
```

### 2.3 设备绑定流程

```
首次登录 → 检测设备指纹
     ↓
┌──────────────────┬────────────────────┐
│ Windows: CPU ID  │ 备用: 系统UUID     │
│ + 主板序列号     │ 或 随机UUID        │
└──────────────────┴────────────────────┘
     ↓
SHA256 哈希 → 生成设备指纹 → 绑定到卡密
```

### 4.1 LicenseManager 核心类

```python
from __future__ import annotations

import hashlib
import json
import random
import subprocess
import threading
import time
import uuid
from datetime import datetime
from typing import Any, Optional
from urllib.parse import urlencode

import httpx
from PySide6.QtCore import QObject, QThread, Signal


error_codes = {
    "-10002": "卡密已过期",
    "-10013": "卡密已被禁用",
    "-10014": "卡密不存在",
    "-10034": "应用不可用",
    "-10035": "用户不在线",
    "-10038": "应用已停用",
    "-10045": "授权已失效",
}


class HeartbeatThread(QThread):
    status_changed = Signal(str)
    invalidated = Signal(str)

    def __init__(self, manager: "LicenseManager"):
        super().__init__()
        self.manager = manager
        self._stop_event = threading.Event()

    def stop(self) -> None:
        self._stop_event.set()

    def run(self) -> None:
        while not self._stop_event.is_set() and self.manager.is_logged_in:
            start_time = time.time()
            keep_running, message = self.manager.perform_heartbeat()
            self.status_changed.emit(message)
            if not keep_running:
                self.invalidated.emit(message)
                break

            elapsed = time.time() - start_time
            wait_time = max(0, self.manager.options.heart_interval_sec - elapsed)
            end_time = time.time() + wait_time
            while time.time() < end_time:
                if self._stop_event.is_set():
                    return
                time.sleep(0.2)


class LicenseManager(QObject):
    license_invalidated = Signal(str)
    heartbeat_status = Signal(str)
    card_info_updated = Signal(object)

    def __init__(self, options):
        super().__init__()
        self.options = options
        self.state = self._load_state()
        self.token = ""
        self.card_info: dict[str, Any] = {}
        self.current_card = ""
        self.current_mac = ""
        self.heart_fail_count = 0
        self.heartbeat_thread: Optional[HeartbeatThread] = None

        if self.options.bind_hardware and not self.state.device_fingerprint:
            self.state.device_fingerprint = self._build_device_fingerprint()
            self._save_state()

    @property
    def is_enabled(self) -> bool:
        return self.options.enabled

    @property
    def is_logged_in(self) -> bool:
        return bool(self.token)

    def get_remembered_card(self) -> str:
        if not self.state.remember_card or not self.state.card_ciphertext:
            return ""
        success, plain_text = try_decrypt_text(self.state.card_ciphertext)
        return plain_text if success else ""

    def login(self, card: str, remember_card: Optional[bool] = None) -> tuple[bool, str, Optional[dict]]:
        remember_card = self.options.remember_card_default if remember_card is None else remember_card
        normalized_card = card.strip()
        mac = self.state.device_fingerprint if self.options.bind_hardware and self.state.device_fingerprint else ""
        if self.options.bind_hardware and not mac:
            mac = self._build_device_fingerprint()
        
        success, message, data = self._request("login", {"card": normalized_card, "mac": mac})
        if not success:
            return False, message, None

        self.current_card = normalized_card
        self.current_mac = mac
        self.token = str((data or {}).get("token", ""))
        self.card_info = dict(data or {})
        self.heart_fail_count = 0

        info_success, _, info_data = self.get_card_info()
        if info_success and info_data:
            self.card_info = info_data

        self.state.remember_card = bool(remember_card)
        self.state.card_ciphertext = encrypt_text(normalized_card) if remember_card else ""
        self.state.last_login_at = datetime.now().isoformat(timespec="seconds")
        self.state.last_expire_at = self._extract_expire_text(self.card_info)
        self.state.last_card_info = self.card_info
        self._save_state()
        self.card_info_updated.emit(self.card_info)
        return True, "登录成功", self.card_info

    def get_card_info(self) -> tuple[bool, str, Optional[dict]]:
        if not self.token:
            return False, "登录状态已失效", None
        if not self.current_card:
            remembered_card = self.get_remembered_card().strip()
            if remembered_card:
                self.current_card = remembered_card
        if not self.current_card:
            return False, "缺少卡密信息", None

        success, message, response_dict = self._request_raw(
            "info",
            {"token": self.token, "card": self.current_card, "mac": "mac"},
            signature_in_params=True,
        )
        if not success:
            return False, message, None
        if response_dict.get("code") != 1:
            return False, str(response_dict.get("msg", "获取卡密信息失败")), None

        data = dict(response_dict.get("data", {}) or {})
        self.card_info = data
        self.state.last_expire_at = self._extract_expire_text(self.card_info)
        self.state.last_card_info = self.card_info
        self._save_state()
        self.card_info_updated.emit(self.card_info)
        return True, str(response_dict.get("msg", "获取卡密信息成功")), data

    def logout(self) -> tuple[bool, str]:
        self.stop_heartbeat()
        if not self.token:
            return True, "已退出"
        success, message, _ = self._request("logout", {"token": self.token})
        self._clear_session(keep_saved_card=True)
        return success, message

    def start_heartbeat(self) -> None:
        if not self.is_enabled or not self.is_logged_in:
            return
        if self.heartbeat_thread and self.heartbeat_thread.isRunning():
            return
        self.heartbeat_thread = HeartbeatThread(self)
        self.heartbeat_thread.status_changed.connect(self.heartbeat_status.emit)
        self.heartbeat_thread.invalidated.connect(self._on_heartbeat_invalidated)
        self.heartbeat_thread.start()

    def stop_heartbeat(self) -> None:
        if self.heartbeat_thread and self.heartbeat_thread.isRunning():
            self.heartbeat_thread.stop()
            self.heartbeat_thread.wait(1500)
        self.heartbeat_thread = None

    def perform_heartbeat(self) -> tuple[bool, str]:
        if not self.token:
            return False, "授权已失效"

        success, message, response_dict = self._request_raw("heart", {"token": self.token})
        if not success:
            self.heart_fail_count += 1
            retry_message = f"心跳异常：{message}（{self.heart_fail_count}/{self.options.heart_error_threshold}）"
            if self.heart_fail_count >= self.options.heart_error_threshold:
                self._clear_session(keep_saved_card=True)
                return False, message
            return True, retry_message

        if response_dict.get("code") != 1:
            error_code = str(response_dict.get("code", ""))
            error_message = error_codes.get(error_code, str(response_dict.get("msg", "未知错误")))
            if error_code in {"-10002", "-10013", "-10014", "-10034", "-10038", "-10045"}:
                self._clear_session(keep_saved_card=True)
                return False, error_message

            self.heart_fail_count += 1
            retry_message = f"心跳异常：{error_message}（{self.heart_fail_count}/{self.options.heart_error_threshold}）"
            if self.heart_fail_count >= self.options.heart_error_threshold:
                self._clear_session(keep_saved_card=True)
                return False, error_message
            return True, retry_message

        self.heart_fail_count = 0
        return True, "在线状态：正常"

    def _on_heartbeat_invalidated(self, message: str) -> None:
        self.license_invalidated.emit(message)

    def _clear_session(self, keep_saved_card: bool) -> None:
        self.token = ""
        self.card_info = {}
        self.current_card = ""
        self.current_mac = ""
        self.heart_fail_count = 0
        if not keep_saved_card:
            self.state.remember_card = False
            self.state.card_ciphertext = ""
        self._save_state()

    def _request(self, action: str, business_params: dict) -> tuple[bool, str, Optional[dict]]:
        success, message, response_dict = self._request_raw(action, business_params)
        if not success:
            return False, message, None
        if response_dict.get("code") != 1:
            code = str(response_dict.get("code", ""))
            message = error_codes.get(code, str(response_dict.get("msg", "请求失败")))
            return False, message, None
        return True, str(response_dict.get("msg", "请求成功")), dict(response_dict.get("data", {}) or {})

    def _request_raw(
        self,
        action: str,
        business_params: dict,
        *,
        signature_in_params: bool = False,
    ) -> tuple[bool, str, Optional[dict]]:
        try:
            params = dict(business_params)
            self._extend_params(params)
            query = self._build_query_string(params)
            signature = self._sha256(query + self.options.app_key)
            if signature_in_params:
                params["signature"] = signature
                plain_params = self._build_query_string(params)
            else:
                plain_params = f"{query}&signature={signature}"

            encrypted_params = encrypt_aes_ecb_pkcs5(plain_params, self.options.encrypt_key)
            request_url = f"{self.options.host_url.rstrip('/')}/single/{action}?appId={self.options.app_id}&params={encrypted_params}"

            with httpx.Client(timeout=self.options.request_timeout_sec) as client:
                response = client.get(request_url)

            if response.status_code != 200:
                return False, "网络连接异常", None

            decrypted_response = decrypt_aes_ecb_pkcs5(response.text, self.options.encrypt_key)
            response_dict = json.loads(decrypted_response)
            if not isinstance(response_dict, dict):
                return False, "授权响应格式异常", None
            return True, "", response_dict
        except httpx.TimeoutException:
            return False, "网络请求超时", None
        except httpx.HTTPError:
            return False, "网络连接异常", None
        except Exception:
            return False, "授权请求异常", None

    def _extend_params(self, params: dict) -> None:
        params["timestamp"] = int(time.time() * 1000)
        params["safeCode"] = random.randint(10**15, 10**16 - 1)

    def _build_query_string(self, params: dict) -> str:
        return urlencode(dict(sorted(params.items())))

    def _sha256(self, text: str) -> str:
        return hashlib.sha256(text.encode("utf-8")).hexdigest()

    def _build_device_fingerprint(self) -> str:
        if self.state.device_fingerprint:
            return self.state.device_fingerprint

        components: list[str] = []
        commands = [
            ("CPU", "(Get-CimInstance Win32_Processor).ProcessorId"),
            ("Board", "(Get-CimInstance Win32_BaseBoard).SerialNumber"),
        ]
        for prefix, command in commands:
            value = self._run_powershell(command)
            if value and value not in {"None", "0", "To be filled by O.E.M."}:
                components.append(f"{prefix}:{value}")

        if not components:
            system_uuid = self._run_powershell("(Get-CimInstance Win32_ComputerSystemProduct).UUID")
            if system_uuid and system_uuid != "UUID":
                components.append(f"UUID:{system_uuid}")

        if not components:
            if not self.state.fallback_device_id:
                self.state.fallback_device_id = str(uuid.uuid4())
                self._save_state()
            components.append(f"FALLBACK:{self.state.fallback_device_id}")

        raw_text = "_".join(components)
        fingerprint = hashlib.sha256(raw_text.encode("utf-8")).hexdigest()
        self.state.device_fingerprint = fingerprint
        self._save_state()
        return fingerprint

    def _run_powershell(self, command: str) -> str:
        try:
            result = subprocess.run(
                ["powershell", "-Command", command],
                capture_output=True,
                text=True,
                timeout=4,
                check=False,
            )
            return result.stdout.strip()
        except Exception:
            return ""

    def _extract_expire_text(self, info: dict) -> str:
        for key in ("endTime", "expireTime", "deadline", "dueTime"):
            value = info.get(key)
            if value:
                return str(value)
        return ""
```

### 4.2 加解密服务 (crypto_service.py)

```python
from __future__ import annotations

import base64
import hashlib
from typing import Any

from cryptography.fernet import Fernet
from Crypto.Cipher import AES
from Crypto.Util.Padding import pad, unpad


# 本地数据加密种子（可自定义）
config_seed = b"VidPromptClient::Config::v1"
CONFIG_FERNET_KEY = base64.urlsafe_b64encode(hashlib.sha256(config_seed).digest())


def get_fernet() -> Fernet:
    return Fernet(CONFIG_FERNET_KEY)


def encrypt_text(plain_text: str) -> str:
    token = get_fernet().encrypt(plain_text.encode("utf-8"))
    return token.decode("utf-8")


def decrypt_text(cipher_text: str) -> str:
    text = get_fernet().decrypt(cipher_text.encode("utf-8"))
    return text.decode("utf-8")


def try_decrypt_text(cipher_text: str) -> tuple[bool, str]:
    try:
        return True, decrypt_text(cipher_text)
    except Exception:
        return False, ""


def encrypt_config_payload(payload: dict[str, Any]) -> bytes:
    import json
    raw_text = json.dumps(payload, ensure_ascii=False, indent=2)
    return get_fernet().encrypt(raw_text.encode("utf-8"))


def decrypt_config_payload(cipher_bytes: bytes) -> dict[str, Any]:
    import json
    plain_text = get_fernet().decrypt(cipher_bytes).decode("utf-8")
    return json.loads(plain_text)


# 卡密平台通信加密（AES ECB PKCS5）
def encrypt_aes_ecb_pkcs5(input_string: str, key: str) -> str:
    key_bytes = key.encode("utf-8")
    if len(key_bytes) not in {16, 24, 32}:
        raise ValueError("encrypt_key 长度必须为 16、24 或 32 字节")
    cipher = AES.new(key_bytes, AES.MODE_ECB)
    encrypted_bytes = cipher.encrypt(pad(input_string.encode("utf-8"), AES.block_size))
    return encrypted_bytes.hex()


def decrypt_aes_ecb_pkcs5(encrypted_hex: str, key: str) -> str:
    key_bytes = key.encode("utf-8")
    if len(key_bytes) not in {16, 24, 32}:
        raise ValueError("encrypt_key 长度必须为 16、24 或 32 字节")
    cipher = AES.new(key_bytes, AES.MODE_ECB)
    decrypted = cipher.decrypt(bytes.fromhex(encrypted_hex))
    return unpad(decrypted, AES.block_size).decode("utf-8")
```

```python
from __future__ import annotations

from dataclasses import asdict, dataclass, field

@dataclass

    device_fingerprint: str = ""
    fallback_device_id: str = ""
    last_login_at: str = ""
```

### 4.4 本地存储 (file_store.py)

```python
from __future__ import annotations

import json
import os
import sys
from pathlib import Path
from typing import Any


client_app_name = "VidPromptClient"


def ensure_dir(path: Path) -> Path:
    path.mkdir(parents=True, exist_ok=True)
    return path


def get_appdata_root() -> Path:
    appdata = os.getenv("APPDATA")
    if appdata:
        return Path(appdata)
    return ensure_dir(Path.home() / "AppData" / "Roaming")


def get_client_data_dir() -> Path:
    return ensure_dir(get_appdata_root() / client_app_name)


def get_license_state_file() -> Path:
    return get_client_data_dir() / "license_state.json"


def get_client_config_file() -> Path:
    if getattr(sys, "frozen", False):
        return Path(sys.executable).resolve().parent / "app_config.enc"
    return Path.cwd() / "app_config.enc"


def read_json_file(path: Path, default: Any) -> Any:
    try:
        with path.open("r", encoding="utf-8") as file:
            return json.load(file)
    except FileNotFoundError:
        return default
    except json.JSONDecodeError:
        return default


def write_json_file(path: Path, data: Any) -> None:
    ensure_dir(path.parent)
    temp_path = path.with_suffix(path.suffix + ".tmp")
    with temp_path.open("w", encoding="utf-8") as file:
        json.dump(data, file, ensure_ascii=False, indent=2)
    temp_path.replace(path)


def read_bytes_file(path: Path, default: bytes = b"") -> bytes:
    try:
        return path.read_bytes()
    except FileNotFoundError:
        return default


def write_bytes_file(path: Path, content: bytes) -> None:
    ensure_dir(path.parent)
    temp_path = path.with_suffix(path.suffix + ".tmp")
    temp_path.write_bytes(content)
    temp_path.replace(path)
```

```python
        self.status_label = QLabel("请输入卡密后登录")

        self.button_box = QDialogButtonBox(
            QDialogButtonBox.StandardButton.Ok | QDialogButtonBox.StandardButton.Cancel
        )
        self.login_button = self.button_box.button(QDialogButtonBox.StandardButton.Ok)
        self.cancel_button = self.button_box.button(QDialogButtonBox.StandardButton.Cancel)
        self.login_button.setText("登录")
        self.cancel_button.setText("取消")

        form_layout.addWidget(card_title)
        form_layout.addWidget(card_hint)
        form_layout.addLayout(form)
        form_layout.addWidget(self.status_label)
        form_layout.addWidget(self.button_box)
        layout.addWidget(form_card)

        self.card_edit.returnPressed.connect(self._handle_login)
        self.button_box.accepted.connect(self._handle_login)
        self.button_box.rejected.connect(self.reject)

    def _load_default_values(self) -> None:
        remembered_card = self.license_manager.get_remembered_card()
        self.remember_check.setChecked(self.license_manager.options.remember_card_default)
        if remembered_card:
            self.card_edit.setText(remembered_card)
            self._set_status_message("检测到已保存卡密，正在尝试自动登录...", "info")
            QTimer.singleShot(80, self._handle_login)

    def _set_loading(self, loading: bool) -> None:
        self.card_edit.setEnabled(not loading)
        self.remember_check.setEnabled(not loading)
        self.login_button.setEnabled(not loading)
        self.cancel_button.setEnabled(not loading)
        self.login_button.setText("登录中..." if loading else "登录")

    def _set_status_message(self, message: str, state: str) -> None:
        self.status_label.setText(message)
        self.status_label.setProperty("state", state)

    def _handle_login(self) -> None:
        card = self.card_edit.text().strip()
        if not card:
            self._set_status_message("请输入卡密后再登录。", "warning")
            self.card_edit.setFocus()
            return

        self._set_loading(True)
        self._set_status_message("正在登录授权平台，请稍候...", "info")
        # 异步执行登录
        self.worker = CallableWorker(
            self.license_manager.login, 
            card, 
            self.remember_check.isChecked()
        )
        self.worker.succeeded.connect(self._on_login_finished)
        self.worker.failed.connect(self._on_login_failed)
        self.worker.start()

    def _on_login_finished(self, result: object) -> None:
        self._set_loading(False)
        success, message, _ = result
        if success:
            self._set_status_message(message or "登录成功", "success")
            self.accept()
            return
        self._set_status_message(message or "登录失败，请检查卡密。", "error")
        QMessageBox.warning(self, "登录失败", message)

    def _on_login_failed(self, message: str) -> None:
        self._set_loading(False)
        self._set_status_message(message, "error")
        QMessageBox.critical(self, "登录异常", message)
```

### 4.6 异步工作线程 (workers.py)

```python
from __future__ import annotations

from typing import Any, Callable

from PySide6.QtCore import QThread, Signal


class CallableWorker(QThread):
    succeeded = Signal(object)
    failed = Signal(str)

    def __init__(self, fn: Callable[..., Any], *args: Any, **kwargs: Any):
        super().__init__()
        self.fn = fn
        self.args = args
        self.kwargs = kwargs

    def run(self) -> None:
        try:
            result = self.fn(*self.args, **self.kwargs)
            self.succeeded.emit(result)
        except Exception as exc:
            self.failed.emit(str(exc))
```

## 五、请求签名算法

### 5.1 签名步骤

1. **构建业务参数**：`{"card": "卡密", "mac": "设备指纹"}`
2. **扩展参数**：添加 `timestamp` 和 `safeCode`
3. **排序并拼接**：按 key 字母序排序，生成 `key1=value1&key2=value2`
4. **计算签名**：`SHA256(拼接串 + app_key)`
5. **加密参数**：AES ECB PKCS5 加密
6. **发送请求**：`GET /single/{action}?appId={app_id}&params={加密参数}`

### 5.2 签名示例

```python
# 1. 业务参数
params = {"card": "TEST123", "mac": "abc123"}

# 2. 扩展参数
params["timestamp"] = 1700000000000
params["safeCode"] = 1234567890123456

# 3. 排序拼接
query = urlencode(sorted(params.items()))
# 结果: card=TEST123&mac=abc123&safeCode=1234567890123456&timestamp=1700000000000

# 4. 计算签名
signature = sha256(query + app_key)

# 5. 拼接签名
plain_params = f"{query}&signature={signature}"

# 6. AES 加密
encrypted = encrypt_aes_ecb_pkcs5(plain_params, encrypt_key)
```

### 配置步骤

3. 在主程序中集成登录对话框

### 注意事项

- `encrypt_key` 必须为 16、24 或 32 字节
- 心跳间隔建议 30-60 秒
- 设备指纹首次生成后绑定，更换设备需重新登录
- 本地存储的卡密使用 Fernet 加密保护
